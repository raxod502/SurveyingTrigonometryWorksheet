  \emph{Recall:} For a right triangle,
  \begin{center}
    \begin{tikzpicture}
      \draw (0, 0) -- (5, 0) node [midway, below=0.08] {$A$} -- (5, 2.5)
      node [midway, right=0.04] {$B$} -- cycle node [midway, above left]
      {$C$};

      \begin{scope}
        \clip (0, 0) -- (5, 0) -- (5, 2.5) -- cycle;

        \draw (5, 0) rectangle ++(-0.3, 0.3);

        \draw (0, 0) circle (0.5) node [above right=0.03 and 0.8] {$u$};

        \draw (5, 2.5) circle (0.3) node [below left=0.28 and 0.03]
        {$v$};
      \end{scope}
    \end{tikzpicture}
  \end{center}
  sine and cosine relate the angles and sides as follows:
  \begin{align*}
    \cos u = \sin v &= \frac{A}{C} \\
    \sin u = \cos v &= \frac{B}{C},
  \end{align*}
  or equivalently
  \begin{align*}
    B &= C \cos u = C \sin v \\
    A &= C \sin u = C \cos v.
  \end{align*}
\end{mdframed}
